
app.get('/api/books', (req, res) => {
  let result = books;

  const { author, year } = req.query;

  if (author) {
    result = result.filter(b =>
      b.author.toLowerCase().includes(author.toLowerCase())
    );
  }

  if (year) {
    result = result.filter(b => b.year == year);
  }

  res.json(result);
});
